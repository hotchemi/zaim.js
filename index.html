Zaim.js

Node.js library for the Zaim API.

Install

Install from npm:

$ npm install zaim

Usage

このドキュメントは編集中です｡各メソッドに渡すパラメータは本家 (https://zaim.net/home/api) に準拠しています｡
お急ぎの方はlib/zaim.jsとtest/以下を見てくれれば理解できると思われます｡

Constructor

var Zaim = require('../lib/zaim');

var zaim = new Zaim({
  // must configure cossumer key and secret
  consumerKey: 'CONSUMER KEY',
  consumerSecret: 'CONSUMER SECRET',
  // option params
  accessToken: 'ACCESS TOKEN',
  accessTokenSecret: 'ACCESS TOKEN SECRET',
  callbackUrl: 'CALLBACK URL'
});

Authorization url

// return https://www.zaim.net/users/auth?oauth_token=***
var authUrl = zaim.getAuthorizationUrl();

Set token and secret

zaim.setAccessToken('accessToken');
zaim.setAccessTokenSecret('accessTokenSecret');

User

zaim.getCredentials(function(data, err) {
  console.log(data);
});

Create payment

zaim.createPay(function(data, err){},
  { category_id: 'category_id',
    genre_id: 'genre_id',
    price: 100,
    date: '2013-04-10' });

Create income

zaim.createIncome(function(data, err){},
  { category_id: 'category_id',
    price: 100,
    date: '2013-04-10' });

Get money

zaim.getMoney(function(data, err) {
  console.log(data);
});

Get payment categories

zaim.getPayCategories(function(data, err) {
  console.log(data);
});

Get payment categories

zaim.getPayCategories(function(data, err) {
  console.log(data);
});

Get income categories

zaim.getIncomeCategories(function(data, err) {
  console.log(data);
});

Get payment genres

zaim.getPayGenres(function(data, err) {
  console.log(data);
}, {lang: 'ja'});

Get currencies

zaim.getCurrencies(function(data, err) {
  console.log(data);
});
